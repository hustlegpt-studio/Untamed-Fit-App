====================================================
STEP 5 — NOTIFICATION FOR KEVIN
====================================================

Create:
  /utils/notifications.ts

Add:

export function addNotification(note) {
  const notes = JSON.parse(localStorage.getItem("untamedNotifications") || "[]");
  notes.push(note);
  localStorage.setItem("untamedNotifications", JSON.stringify(notes));
}

When a trainee books:
addNotification({
  id: crypto.randomUUID(),
  type: "booking",
  message: `${traineeEmail} booked a session on ${date} at ${time}.`,
  createdAt: new Date().toISOString()
});
