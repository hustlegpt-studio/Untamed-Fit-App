export function createUser(email, password) {
  const users = JSON.parse(localStorage.getItem("untamedUsers") || "[]");
  users.push({ email, password, profile: {} });
  localStorage.setItem("untamedUsers", JSON.stringify(users));
}

export function loginUser(email, password) {
  const users = JSON.parse(localStorage.getItem("untamedUsers") || "[]");
  return users.find(u => u.email === email && u.password === password) || null;
}

export function setActiveUser(email) {
  localStorage.setItem("untamedActiveUser", email);
}

export function getActiveUser() {
  return localStorage.getItem("untamedActiveUser");
}
